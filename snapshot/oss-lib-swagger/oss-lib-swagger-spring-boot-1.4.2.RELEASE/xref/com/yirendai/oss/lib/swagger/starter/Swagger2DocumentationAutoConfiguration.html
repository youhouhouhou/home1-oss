93    @ConditionalOnMissingBean(name = DOCKET_APPLICATION)
94    @Bean(name = DOCKET_APPLICATION)
95    public Docket docketApplication() {
96      final Docket docket = new Docket(DocumentationType.SWAGGER_2) //
97        .apiInfo(apiInfo("application", "application's endpoints")) //
98        .groupName("api") //
99        .select() //
100       .apis(or(this.applicationAips(), this.springApis())) //
101       .paths(not(this.managementPaths)) //
102       .build();
103     final Optional<ResolvedType> modelResolvedError = modelResolvedError();
104     return modelResolvedError.isPresent() ? docket.additionalModels(modelResolvedError.get()) : docket;
105   }
106

115   private Predicate<RequestHandler> applicationAips() {
116     final Predicate<RequestHandler> ossLibApi = RequestHandlerSelectors.basePackage("com.yirendai"); // TODO 从此类包取前两段
117     final String applicationPackage = AppUtils.appBasePackage("");
118     return isNotBlank(applicationPackage) ? //
119       or(RequestHandlerSelectors.basePackage(applicationPackage), ossLibApi) : //
120       RequestHandlerSelectors.any();

161       private RequestMatcher swaggerRequestMatcher() {
162         final Set<String> paths = newLinkedHashSet();
163         // default is: /v2/api-docs
164         paths.add("/webjars/**");
165         paths.add(this.environment.getProperty("springfox.documentation.swagger.v1.path", "/v1/**"));
166         paths.add(this.environment.getProperty("springfox.documentation.swagger.v2.path", "/v2/**"));
167         paths.addAll(newArrayList("/swagger-ui.html", "/swagger-resources/**"));
168 
169         final RequestMatcher requestMatcher;
170         if (!paths.isEmpty()) {
171           final List<RequestMatcher> matchers = paths.stream() //
172             .map(pattern -> new AntPathRequestMatcher(pattern, null)) //
173             .collect(toList());
174           requestMatcher = new OrRequestMatcher(matchers);
175         } else {
176           requestMatcher = null;
177         }
178         return requestMatcher;
179       }
180     }
