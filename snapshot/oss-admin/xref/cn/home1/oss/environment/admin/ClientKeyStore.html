3   import java.util.Collection;
4   import java.util.concurrent.ConcurrentHashMap;
5   import java.util.concurrent.ConcurrentMap;

10  public class ClientKeyStore {
11  
12    private final ConcurrentMap<String, ClientKey> keysMap = new ConcurrentHashMap<>();
13  
14  
15    public ClientKey save(ClientKey clientKey) {
16      return keysMap.put(clientKey.getServiceId(), clientKey);
17    }
18  
19    public Collection<ClientKey> findAll() {
20      return keysMap.values();
21    }
22  
23    public ClientKey find(String serviceId) {
24      return keysMap.get(serviceId);
25    }
26  
27    public ClientKey delete(String serviceId) {
28      return keysMap.remove(serviceId);
29    }
30  
31    public boolean isAvailable(String serviceId) {
32      ClientKey clientKey = find(serviceId);
33      return (clientKey == null) ? false : true;
34    }
35  }
