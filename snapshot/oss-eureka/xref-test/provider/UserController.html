4   import org.springframework.beans.factory.annotation.Autowired;
5   import org.springframework.cloud.client.ServiceInstance;
6   import org.springframework.cloud.client.discovery.DiscoveryClient;
7   import org.springframework.web.bind.annotation.GetMapping;
8   import org.springframework.web.bind.annotation.PathVariable;
9   import org.springframework.web.bind.annotation.RestController;
10  
11  /**
12   * ① 测试服务实例的相关内容. 
13   * ② 为后来的服务做提供.
14   */
15  @RestController
16  public class UserController {
17    @Autowired
18    private DiscoveryClient discoveryClient;
19    @Autowired
20    private UserRepository userRepository;
21  
22    /**
23     * 注：@GetMapping("/{id}")是spring 4.3的新注解等价于：
24     * 
25     * @RequestMapping(value = "/id", method = RequestMethod.GET)
26     *                       类似的注解还有@PostMapping等等
27     * @param id - 用户id
28     * @return user信息
29     */
30    @GetMapping("/{id}")
31    public User findById(@PathVariable final Long id) {
32      return this.userRepository.findOne(id);
33    }
34  
35    /**
36     * 本地服务实例的信息.
37     * 
38     * @return 服务Instance
39     */
40    @GetMapping("/instance-info")
41    public ServiceInstance showInfo() {
42      return this.discoveryClient.getLocalServiceInstance();
43    }
44  }
