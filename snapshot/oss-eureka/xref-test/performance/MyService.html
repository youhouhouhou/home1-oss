22  /**
23   * 目的：
24   * 创建N个Instance链接指定EurekaServer
25   * 运行举例：
26   * java performance.MyService http://local-eureka:8761/eureka/ 100
27   * 第一个参数：EurekaServer注册地址
28   * 第二个参数：创建Instance个数，默认1个
29   * 按 回车 退出程序
30   *
31   * @author zhupengcheng
32   */
33  public class MyService {
34    private static final int ONE = 1;
35  
36    public static void main(final String... args) {
37      String eurekaUrl = "http://local-eureka:8761/eureka/";
38      if (args.length > 0) {
39        eurekaUrl = args[0];
40      }
41  
42      int connectNum = 1;
43      if (args.length > ONE) {
44        connectNum = Integer.parseInt(args[1]);
45      }
46  
47      System.err.println("eurekaServer:" + eurekaUrl + "  instance总数：" + connectNum);
48  
49      final List<EurekaClient> clientList = new ArrayList<EurekaClient>();
50  
51      for (int i = 0; i < connectNum; i++) {
52        final AbstractConfiguration abstractConfiguration = ConfigurationManager.getConfigInstance();
53  
54        final String appName = "instance-" + i;
55        final String port = "801" + i;
56  
57        abstractConfiguration.setProperty("eureka.region", "default");
58        abstractConfiguration.setProperty("eureka.name", appName);
59        abstractConfiguration.setProperty("eureka.vipAddress", "sampleservice.mydomain.ne");
60        abstractConfiguration.setProperty("eureka.port", port);
61        abstractConfiguration.setProperty("eureka.preferSameZone", "true");
62        abstractConfiguration.setProperty("eureka.shouldUseDns", "false");
63        abstractConfiguration.setProperty("eureka.serviceUrl.default", eurekaUrl);
64  
65        com.netflix.config.DynamicPropertyFactory.getInstance();
66  
67        final InstanceInfo.Builder builder = InstanceInfo.Builder.newBuilder();
68        builder.setAppName(appName);
69        builder.setInstanceId(appName);
70        builder.setPort(Integer.parseInt(port));
71        builder.setLeaseInfo(LeaseInfo.Builder.newBuilder().build());
72        builder.setDataCenterInfo(getDataCenterInfo());
73        // url 随便写，不写的话在Server查看页面看不到链接而已
74        builder.setHomePageUrlForDeser("http://www.sina.com.cn");
75        builder.setStatusPageUrlForDeser("http://www.sina.com.cn");
76        builder.setHealthCheckUrlsForDeser("http://www.sina.com.cn", null);
77  
78        final InstanceInfo instanceInfo = builder.build();
79  
80        final MyDataCenterInstanceConfig myConfig = new MyDataCenterInstanceConfig();
81  
82        final DefaultEurekaClientConfig clientConfig = new DefaultEurekaClientConfig();
83  
84        // setup config in advance, used in initialize converter
85        final ApplicationInfoManager applicationInfoManager = new ApplicationInfoManager(myConfig, instanceInfo);
86  
87        final EurekaClient eurekaClient = new DiscoveryClient(applicationInfoManager, clientConfig);
88        applicationInfoManager.setInstanceStatus(InstanceStatus.UP);
89  
90        clientList.add(eurekaClient);
91      }
92      System.out.println("connectNum:" + connectNum + " succ");
93      final Scanner sc = new Scanner(System.in, UTF_8.name());
94      sc.nextLine();
95      System.out.println("Exit Test");
96  
97      for (final EurekaClient eurekaClient : clientList) {
98        eurekaClient.shutdown();
99      }
100     sc.close();
101   }
102 
103   protected static DataCenterInfo getDataCenterInfo() {
104     return new DataCenterInfo() {
105       @Override
106       public Name getName() {
107         return Name.MyOwn;
108       }
109     };
110   }
111 }
