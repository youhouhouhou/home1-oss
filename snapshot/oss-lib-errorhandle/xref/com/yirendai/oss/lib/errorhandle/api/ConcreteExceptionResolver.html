25  public interface ConcreteExceptionResolver<T extends Throwable> extends ExceptionResolver<T> {
26  
27    Class<T> getExceptionClass();
28  
29    /**
30     * set conversionService.
31     *
32     * @param conversionService convert value in validation error into string
33     */
34    void setConversionService(ConversionService conversionService);
35  
36    ExceptionTranslator getExceptionTranslator();
37  
38    void setExceptionTranslator(ExceptionTranslator exceptionTranslator);
39  
40    void setStackTraceIndicator(StackTraceIndicator stackTraceIndicator);
41  
42    default Optional<HttpHeaders> createHeaders(final RequestAttributes request, final T throwable) {
43      return Optional.of(ResolvedError.newHttpHeaders());
44    }
45  
46    /**
47     * find a request attribute.
48     *
49     * @param requestAttributes requestAttributes
50     * @param name              name
51     * @param <A>               attributeType
52     * @return request attribute
53     */
54    @SuppressWarnings("unchecked")
55    default <A> A getAttribute(final RequestAttributes requestAttributes, final String name) {
56      return (A) requestAttributes.getAttribute(name, RequestAttributes.SCOPE_REQUEST);
57    }
58  
59    default <E extends Throwable> String message( //
60      final RequestAttributes requestAttributes, //
61      final E throwable, //
62      final Optional<List<ValidationError>> errorsOptional //
63    ) {
64      final Object fromAttribute = getAttribute(requestAttributes, "javax.servlet.error.message");
65      final String fromError;
66      if (throwable != null) {
67        if (!errorsOptional.isPresent()) {
68          fromError = throwable.getMessage();
69        } else {
70          final int errorCount = errorsOptional.get().size();
71          fromError = errorCount > 0 ? "Validation failed. Error count: " + errorCount : "No errors";
72        }
73      } else {
74        fromError = null;
75      }
76      final String result;
77      if ((!isEmpty(fromAttribute) || fromError == null) && !(throwable instanceof BindingResult)) {
78        result = isEmpty(fromAttribute) ? "No message available" : fromAttribute.toString();
79      } else {
80        result = fromError;
81      }
82      return result;
83    }
84  
85    default Optional<Location> find(final T throwable) {
86      final ExceptionTranslator exceptionTranslator = this.getExceptionTranslator();
87      return exceptionTranslator != null ? exceptionTranslator.find(throwable) : Optional.empty();
88    }
89  
90    default Optional<Integer> status( //
91      final RequestAttributes requestAttributes, //
92      final Location location, //
93      final T throwable //
94    ) {
95      final ExceptionTranslator exceptionTranslator = this.getExceptionTranslator();
96      final Optional<Integer> translated = exceptionTranslator != null ? //
97        exceptionTranslator.status(location) : //
98        Optional.empty();
99      final Integer fromAttribute = this.getAttribute(requestAttributes, "javax.servlet.error.status_code");
100     return translated.isPresent() ? translated : Optional.ofNullable(fromAttribute);
101   }
102 
103   /**
104    * Translate exception and generates localizedMessage.
105    *
106    * @param request   The current request.
107    * @param location  The location get data from.
108    * @param throwable throwable
109    * @return A TranslateResult.
110    */
111   default Optional<String> localizedMessage( //
112     final RequestAttributes request, //
113     final Location location, //
114     final T throwable //
115   ) {
116     final ExceptionTranslator exceptionTranslator = this.getExceptionTranslator();
117     final Optional<String> result;
118     if (exceptionTranslator != null) {
119       final Map<String, Serializable> contextVariables = ImmutableMap.of();
120       final String template = exceptionTranslator.template(location).orElse(null);
121       result = exceptionTranslator.localizedMessage(template, request, throwable, contextVariables);
122     } else {
123       result = Optional.empty();
124     }
125     return result;
126   }
127 
128   /**
129    * validation errors.
130    *
131    * @param throwable {@link ConstraintViolationException}, {@link MethodArgumentNotValidException}
132    * @return validation errors
133    */
134   default Optional<List<ValidationError>> validationErrors(final T throwable) {
135     return Optional.empty();
136   }
137 }
