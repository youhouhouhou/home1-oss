35   * 经过解析的错误信息.
36   *
37   * <p>
38   * Created by zhanghaolun on 16/7/1.
39   * </p>

41  @XmlRootElement(name = "error") // Jaxb2RootElementHttpMessageConverter
42  @XmlAccessorType(XmlAccessType.FIELD)
43  @XmlType(propOrder = {"error", "exception", "errors", "message", "path", "status", "timestamp", "trace", //
44    "datetime", "headers", "localizedMessage", "tracks"})
45  @JsonInclude(JsonInclude.Include.NON_EMPTY) // for Jackson 2.x
46  @JsonSerialize(include = JsonSerialize.Inclusion.NON_EMPTY) // for Jackson 1.x
47  @Builder(builderMethodName = "resolvedErrorBuilder")
48  @AllArgsConstructor(access = PRIVATE)
49  @ToString
50  @EqualsAndHashCode(of = {"error", "exception", "path", "status", "timestamp", "localizedMessage"})
51  @Setter(PRIVATE)
52  @Getter
53  @Slf4j
54  public class ResolvedError implements Serializable {
55  
56    public static final String RESOLVED_ERROR_COOKIE = "resolvedErrorCookie";
57    public static final String RESOLVED_ERROR_OBJECT = "resolvedError";
58    static final String HEADER_RESOLVED_ERROR = "RESOLVED-ERROR";
59    private static final long serialVersionUID = 1L;
60  
61    // ------------------------------ basic ------------------------------
62    @JsonProperty("error")
63    private String error;
64    /**
65     * 解析得到的 数据验证错误信息.
66     * Nested XmlElements see: https://github.com/FasterXML/jackson-module-jaxb-annotations/issues/42
67     */
68    @JsonProperty("errors")
69    @XmlElementWrapper(name = "errors")
70    @XmlElements(@XmlElement(name = "error"))
71    private ValidationError[] errors;
72    @XmlElement
73    private String exception;
74    private String message;
75    private String path;
76    private Integer status;
77    private Long timestamp;
78    private String trace;
79    // ------------------------------ extended ------------------------------
80    /**
81     * ISO8601 string.
82     */
83    private String datetime;
84    /**
85     * 解析得到的 异常响应头信息.
86     */
87    @JsonProperty("headers")
88    @XmlElementWrapper(name = "headers")
89    @XmlElement(name = "header")
90    private HttpHeader[] headers;
91    /**
92     * 解析得到的 错误信息.
93     */
94    private String localizedMessage;
95    /**
96     * 解析得到的 异常的调用路径 (RPC).
97     */
98    @JsonProperty("tracks")
99    @XmlElementWrapper(name = "tracks")
100   @XmlElement(name = "track")
101   private String[] tracks;
102 
103   private ResolvedError() {
104     this.headers = HttpHeader.fromHttpHeaders(newHttpHeaders());
105   }
106 
107   public static HttpHeaders newHttpHeaders() {
108     final HttpHeaders headers = new HttpHeaders();
109     headers.add(HEADER_RESOLVED_ERROR, HEADER_RESOLVED_ERROR);
110     return headers;
111   }
112 
113   public static ResolvedError fromErrorAttributes(final Map<String, Object> map) {
114     return map == null ? null : ResolvedError.resolvedErrorBuilder()
115       // basic
116       .error((String) map.get("error")) //
117       .errors((ValidationError[]) map.get("errors")) //
118       .exception((String) map.get("exception"))
119       .message((String) map.get("message")) //
120       .path((String) map.get("path")) //
121       .status((Integer) map.get("status")) //
122       .timestamp((Long) map.get("timestamp")) //
123       .trace((String) map.get("trace")) //
124       // extended
125       .datetime((String) map.get("datetime"))
126       .headers((HttpHeader[]) map.get("headers")) //
127       .localizedMessage((String) map.get("localizedMessage")) //
128       .tracks((String[]) map.get("tracks")).build();
129   }
130 
131   @JsonIgnore
132   @org.codehaus.jackson.annotate.JsonIgnore
133   public HttpStatus getHttpStatus() {
134     HttpStatus result;
135     try {
136       result = HttpStatus.valueOf(this.status);
137     } catch (final Exception ex) {
138       result = HttpStatus.INTERNAL_SERVER_ERROR;
139       log.debug("error parse http status {}", this.status, ex);
140     }
141     return result;
142   }
143 
144   /**
145    * before set into cookie, call this method to avoid header size exceed limit.
146    *
147    * @return this
148    */
149   public ResolvedError eraseTraces() {
150     this.setTracks(null);
151     this.setTrace(null);
152     return this;
153   }
154 
155   public ResolvedError trackPrepend(final String track) {
156     this.tracks = this.tracks != null ? //
157       ArrayUtils.add(this.tracks, 0, track) : //
158       new String[]{track};
159     return this;
160   }
161 
162   public Map<String, Object> toErrorAttributes() {
163     final Map<String, Object> map = newLinkedHashMap();
164     // basic
165     map.put("error", this.error);
166     map.put("errors", this.errors);
167     map.put("exception", this.exception);
168     map.put("message", this.message);
169     map.put("path", this.path);
170     map.put("status", this.status);
171     map.put("timestamp", this.timestamp);
172     map.put("trace", this.trace);
173     // extended
174     map.put("datetime", datetime);
175     map.put("headers", this.headers);
176     map.put("localizedMessage", this.localizedMessage);
177     map.put("tracks", this.tracks);
178 
179     map.entrySet().removeIf(e -> e.getValue() == null);
180     return map;
181   }
