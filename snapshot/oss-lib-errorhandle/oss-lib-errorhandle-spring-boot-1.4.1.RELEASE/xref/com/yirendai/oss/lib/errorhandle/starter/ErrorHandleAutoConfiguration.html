203   public static ExceptionTranslator buildExceptionTranslator(final AppErrorProperties.SearchStrategy searchStrategy) {
204     final Optional<MessageSource> messageSourceOptional = messageSource("classpath:/errorhandle/default");
205     final MessageSource defaultMessageSource = messageSourceOptional.isPresent() ? messageSourceOptional.get() : null;
206     checkNotNull(defaultMessageSource, "message source classpath:/errorhandle/default not present");
207 
208     final Optional<MessageSource> applicationMessageSource = messageSource("classpath:/errorhandle/application");
209     final List<MessageSource> messageSources = applicationMessageSource.isPresent() ? //
210       newArrayList(applicationMessageSource.get(), defaultMessageSource) : //
211       newArrayList(defaultMessageSource);
212 
213     final Comparator<Location> locationComparator = HIERARCHY_FIRST == searchStrategy ? //
214       HIERARCHY_FIRST_COMPARATOR : ORDER_FIRST_COMPARATOR;
215 
216     final DefaultExceptionTranslator defaultExceptionTranslator = new DefaultExceptionTranslator();
217     defaultExceptionTranslator.setLocationComparator(locationComparator);
218     defaultExceptionTranslator.setMessageSources(messageSources);
219     defaultExceptionTranslator.setMessageInterpolator(new SpelMessageInterpolator());
220     return defaultExceptionTranslator;
221   }

223   public static TypeSafeCookie<ResolvedError> buildResolvedErrorCookie( //
224     final DomainResolver domainResolver, final ObjectMapper objectMapper) {
225     final JsonToken<ResolvedError> jsonToken = new JsonToken<>(ResolvedError.class, objectMapper);
226     final UrlEncodedToken<ResolvedError> urlEncodedToken = new UrlEncodedToken<>(jsonToken);
227     final int maxAge = (int) DAYS.toSeconds(1L);
228     return new TokenBasedCookie<>(domainResolver, false, maxAge, "resolved_error", false, urlEncodedToken);
229   }

231   static Optional<MessageSource> messageSource(final String location) {
232     final Optional<MessageSource> result;
233     if (StringUtils.isNotBlank(location)) {
234       final ReloadableResourceBundleMessageSource messages = new ReloadableResourceBundleMessageSource();
235       messages.setBasename(location);
236       messages.setDefaultEncoding("UTF-8");
237       messages.setFallbackToSystemLocale(false);
238       result = Optional.of(messages);
239     } else {
240       log.info("errorhandle messageSource not found at {}", location);
241       result = Optional.empty();
242     }
243     return result;
244   }
