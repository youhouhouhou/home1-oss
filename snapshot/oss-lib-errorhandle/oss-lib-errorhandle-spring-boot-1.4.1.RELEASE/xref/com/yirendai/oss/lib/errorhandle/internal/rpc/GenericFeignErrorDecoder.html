14  import feign.codec.DecodeException;
15  
16  import lombok.Setter;
17  import lombok.extern.slf4j.Slf4j;
18  
19  import org.springframework.beans.factory.annotation.Autowired;
20  import org.springframework.http.HttpHeaders;
21  import org.springframework.http.HttpStatus;
22  import org.springframework.oxm.jaxb.Jaxb2Marshaller;
23  import org.springframework.util.StreamUtils;
24  import org.springframework.web.client.HttpClientErrorException;
25  import org.springframework.web.client.HttpServerErrorException;
26  
27  @Setter
28  @Slf4j
29  public class GenericFeignErrorDecoder implements feign.codec.ErrorDecoder {
30  
31    private final feign.codec.ErrorDecoder delegate = new Default();
32  
33    private final Jaxb2Marshaller jaxb2Marshaller = JaxbUtils.jaxb2Marshaller();
34    @Autowired
35    private ObjectMapper objectMapper;
36  
37    private static HttpHeaders cloneHeaders(final feign.Response response) {
38      final HttpHeaders responseHeaders = new HttpHeaders();
39      response.headers().entrySet().stream()
40          .forEach(entry -> responseHeaders.put(entry.getKey(), newArrayList(entry.getValue())));
41      return responseHeaders;
42    }
43  
44    private static byte[] responseBody(final feign.Response response) {
45      try {
46        return StreamUtils.copyToByteArray(response.body().asInputStream());
47      } catch (final Exception wrapped) {
48        log.info("Failed to process response body.", wrapped);
49        throw new DecodeException("Failed to process response body.", wrapped);
50      }
51    }
52  
53    @Override
54    public Exception decode(final String methodKey, final feign.Response response) {
55      if (log.isDebugEnabled()) {
56        log.debug("decode : {}", methodKey);
57      }
58  
59      final HttpHeaders responseHeaders = cloneHeaders(response);
60      final byte[] responseBody = responseBody(response);
61      final Exception result;
62      if (ResolvedErrorException.isResolvedError(responseHeaders)) {
63        final String info = toEncodedString(responseBody, UTF_8);
64        try {
65          // 如果被调用的服务配置了XmlHttpMessageConverter(使用XStreamMarshaller) 可能导致数据被包裹在<string></string>里面
66          final String json =
67              info.startsWith("<string>") ? JaxbUtils.unmarshal(this.jaxb2Marshaller, info, String.class) : info;
68          result = new ResolvedErrorException(this.objectMapper.readValue(json, ResolvedError.class));
69        } catch (final Exception wrapped) {
70          log.info("Failed to decode resolvedError. info: " + info, wrapped);
71          throw new DecodeException("Failed to decode resolvedError. info: " + info, wrapped);
72        }
73      } else {
74        final HttpStatus statusCode = HttpStatus.valueOf(response.status());
75        final String statusText = response.reason();
76        if (response.status() >= 400 && response.status() <= 499) {
77          result = new HttpClientErrorException( //
78              statusCode, //
79              statusText, //
80              responseHeaders, //
81              responseBody, //
82              null);
83        } else if (response.status() >= 500 && response.status() <= 599) {
84          result = new HttpServerErrorException( //
85              statusCode, //
86              statusText, //
87              responseHeaders, //
88              responseBody, //
89              null);
90        } else {
91          result = delegate.decode(methodKey, response);
92        }
93      }
94      return result;
95    }
96  }
