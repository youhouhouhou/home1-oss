42    }

44    @Test
45    public void testStatusCodeAndMessage() throws Exception {
46      // In a real container the response is carried over into the error dispatcher, but
47      // in the mock a new one is created so we have to assert the status at this
48      // intermediate point
49      final MvcResult result = this.mockMvc.perform(get("/nest"))
50        .andExpect(status().isNotFound()).andReturn();
51  
52      final MvcResult response = this.mockMvc.perform(ErrorhandleMockMvcBuilders.requestToErrorController(result))
53        .andExpect(jsonPath("@.localizedMessage").value("Exception"))
54        .andExpect(jsonPath("message").value("go")).andReturn();
55  
56      // And the rendered status code is always wrong (but would be 400 in a real
57      // system)
58      String content = response.getResponse().getContentAsString();
59      log.info("content:" + content);
60    }
