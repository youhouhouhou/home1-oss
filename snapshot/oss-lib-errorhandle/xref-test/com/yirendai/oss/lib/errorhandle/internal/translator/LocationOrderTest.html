30  @Slf4j
31  public class LocationOrderTest {
32  
33    private ExceptionTranslator hierarchyFirstTranslator;
34    private ExceptionTranslator orderFirstTranslator;
35  
36    @Before
37    public void setUp() {
38      this.hierarchyFirstTranslator = ErrorHandleAutoConfiguration.buildExceptionTranslator(HIERARCHY_FIRST);
39      this.orderFirstTranslator = ErrorHandleAutoConfiguration.buildExceptionTranslator(ORDER_FIRST);
40    }
41  
42    @Test
43    public void testNoHandlerFoundExceptionLocationOrder() {
44      final Throwable throwable = new NoHandlerFoundException("GET", "http://127.0.0.1:8080", new HttpHeaders());
45  
46      final Optional<Location> hierarchyFirst = this.hierarchyFirstTranslator.find(throwable);
47      log.info("hierarchyFirst: {}", hierarchyFirst);
48      assertTrue(hierarchyFirst.isPresent());
49      assertEquals(ImmutableSet.of("classpath:/errorhandle/default"), getSource(hierarchyFirst.get()).getBasenameSet());
50      assertEquals("org.springframework.web.servlet.NoHandlerFoundException", hierarchyFirst.get().getKey());
51      assertEquals(0L, (long) hierarchyFirst.get().getLevel());
52  
53      final Optional<Location> orderFirst = this.orderFirstTranslator.find(throwable);
54      log.info("orderFirst: {}", orderFirst);
55      assertTrue(orderFirst.isPresent());
56      assertEquals(ImmutableSet.of("classpath:/errorhandle/application"), getSource(orderFirst.get()).getBasenameSet());
57      assertEquals("javax.servlet.ServletException", orderFirst.get().getKey());
58      assertEquals(1L, (long) orderFirst.get().getLevel());
59    }
60  
61    @Test
62    public void testConstraintViolationExceptionLocationOrder() {
63      final Throwable throwable = new ConstraintViolationException("empty", ImmutableSet.of());
64  
65      final Optional<Location> hierarchyFirst = this.hierarchyFirstTranslator.find(throwable);
66      assertTrue(hierarchyFirst.isPresent());
67      assertEquals(ImmutableSet.of("classpath:/errorhandle/default"), getSource(hierarchyFirst.get()).getBasenameSet());
68      assertEquals("javax.validation.ConstraintViolationException", hierarchyFirst.get().getKey());
69      assertEquals(0, hierarchyFirst.get().getLevel().intValue());
70  
71      final Optional<Location> orderFirst = this.orderFirstTranslator.find(throwable);
72      log.info("orderFirst: {}", orderFirst);
73      assertTrue(orderFirst.isPresent());
74      assertEquals(ImmutableSet.of("classpath:/errorhandle/application"), getSource(orderFirst.get()).getBasenameSet());
75      assertEquals(DEFAULT_EXCEPTION_CLASS, orderFirst.get().getKey());
76      assertEquals(Integer.MAX_VALUE, orderFirst.get().getLevel().intValue());
77    }
78  
79    private static AbstractResourceBasedMessageSource getSource(final Location location) {
80      return (AbstractResourceBasedMessageSource) location.getSource();
81    }
82  }
