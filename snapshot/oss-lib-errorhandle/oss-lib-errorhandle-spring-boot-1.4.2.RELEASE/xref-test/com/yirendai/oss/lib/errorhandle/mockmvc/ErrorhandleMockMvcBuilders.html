17   * Created by haolun on 16/12/23.

19  @NoArgsConstructor(access = PRIVATE)
20  public abstract class ErrorhandleMockMvcBuilders {
21  
22    public static ResultActions performRequestToErrorController(
23      final MockMvc mockMvc, //
24      final MvcResult originalMvcResult //
25    ) throws Exception {
26      return mockMvc.perform(requestToErrorController(originalMvcResult));
27    }
28  
29    public static RequestBuilder requestToErrorController(final MvcResult originalMvcResult) {
30      return new RequestBuilder() {
31  
32        @Override
33        public MockHttpServletRequest buildRequest(final ServletContext servletContext) {
34          return requestToErrorController("/error", originalMvcResult.getRequest());
35        }
36      };
37    }
38  
39    static MockHttpServletRequest requestToErrorController(
40      final String path,
41      final MockHttpServletRequest originalRequest
42    ) {
43      originalRequest.setDispatcherType(DispatcherType.ERROR);
44      originalRequest.setRequestURI(path);
45      return originalRequest;
46    }
47  }
