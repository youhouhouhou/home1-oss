10  import org.springframework.http.HttpStatus;
11  
12  import java.io.Serializable;
13  import java.util.Map;
14  
15  /**
16   * 应用抛出的通用错误信息.

22  //@JsonInclude(JsonInclude.Include.NON_EMPTY) // for Jackson 2.x
23  //@JsonSerialize(include = JsonSerialize.Inclusion.NON_EMPTY) // for Jackson 1.x
24  //@XmlRootElement(name = "applicationError") // for JAXB
25  @EqualsAndHashCode(callSuper = false, of = {"status", "template"})
26  @Getter
27  public final class ApplicationException extends RuntimeException implements Serializable {
28  
29    private static final long serialVersionUID = 1L;
30  
31    private static final String APPLICATION_EXCEPTION = "application exception";
32  
33    private final HttpStatus status;
34    private final String template;
35    private final Map<String, Serializable> contextVariables;
36  
37    public ApplicationException( //
38      final HttpStatus status, //
39      final String template, //
40      final Map<String, Serializable> contextVariables //
41    ) {
42      super(APPLICATION_EXCEPTION);
43      this.status = checkNotNull(status, "status must not null");
44      this.template = checkNotNull(template, "template must not null");
45      this.contextVariables = checkNotNull(contextVariables, "contextVariables must not null");
46    }
47  
48    public static Boolean isApplicationError(final Throwable throwable) {
49      return throwable != null && ApplicationException.class.isAssignableFrom(throwable.getClass());
50    }
51  
52    public Map<String, Serializable> getContextVariables() {
53      return this.contextVariables;
54    }
55  
56    public HttpStatus getStatus() {
57      return this.status;
58    }
59  
60    public String getTemplate() {
61      return this.template;
62    }
63  
64    @Override
65    public String toString() {
66      final MoreObjects.ToStringHelper toStringHelper = MoreObjects.toStringHelper(ApplicationException.class)
67        .add("status", this.status)
68        .add("template", this.template);
69      if (this.contextVariables != null) {
70        this.contextVariables.entrySet().forEach(entry -> toStringHelper.add(entry.getKey(), entry.getValue()));
71      }
72      return toStringHelper.toString();
73    }
74  }
