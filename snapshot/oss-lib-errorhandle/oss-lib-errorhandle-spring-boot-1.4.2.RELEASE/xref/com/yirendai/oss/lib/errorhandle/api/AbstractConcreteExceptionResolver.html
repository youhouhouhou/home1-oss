63    @Deprecated
64    static HttpStatus parseHttpStatus(final Object value) {
65      Assert.notNull(value, "Values of the resolverMap map must not be null");
66  
67      final HttpStatus result;
68      if (value instanceof HttpStatus) {
69        result = (HttpStatus) value;
70      } else if (value instanceof Integer) {
71        result = HttpStatus.valueOf((int) value);
72      } else if (value instanceof String) {
73        result = HttpStatus.valueOf(Integer.parseInt((String) value));
74      } else {
75        throw new IllegalArgumentException(String.format( //
76          "Values of the resolverMap maps must be instance of " //
77            + "ErrorResponseFactory, HttpStatus, String, or int, " //
78            + "but %s given",
79          value.getClass()));
80      }
81      return result;
82    }

84    @Deprecated
85    @SuppressWarnings("unused")
86    private static <T extends Throwable> Throwable cause(final T exception) {
87      Throwable cause = null;
88      if (exception != null) {
89        cause = exception;
90        while (cause instanceof ServletException && cause.getCause() != null) {
91          final ServletException servletException = ((ServletException) cause);
92          cause = servletException.getCause();
93        }
94      }
95      return cause;
96    }

98    private static String error(final Optional<Integer> statusOptional) {
99      // TODO Optional<Integer>' used as type for parameter
100     String result;
101     if (statusOptional.isPresent()) {
102       final Integer status = statusOptional.get();
103       try {
104         final HttpStatus httpStatus = HttpStatus.valueOf(status);
105         result = httpStatus.getReasonPhrase();
106       } catch (final IllegalArgumentException ignored) { // Unable to obtain a reason
107         if (log.isDebugEnabled()) {
108           log.debug("Unable to obtain a reason, status {}", status, ignored);
109         }
110         result = "Http Status " + status;
111       }
112     } else {
113       result = "None";
114     }
115     return result;
116   }

118   @Override
119   public final ResolvedError resolve( //
120     final HttpServletRequest request, //
121     final T throwable //
122   ) {
123     final DateTime now = Defaults.now();
124 
125     // See http://stackoverflow.com/a/12979543/2217862
126     // This attribute is never set in MockMvc, so it's not covered in integration it.
127     request.removeAttribute(PRODUCIBLE_MEDIA_TYPES_ATTRIBUTE);
128 
129     final RequestAttributes requestAttributes = new ServletRequestAttributes(request);
130 
131     final Boolean stackTrace = this.stackTraceIndicator.stackTrace(request, null);
132     final String path = request.getQueryString() != null ? //
133       request.getRequestURI() + request.getQueryString() : //
134       request.getRequestURI();
135     final String track = stackTrace ? curl(request) : TRACE_OFF;
136 
137     return this.resolve(requestAttributes, throwable, now, stackTrace, path, track);
138   }

146   public final ResolvedError resolve( //

151

153       requestAttributes, //
154       "javax.servlet.error.request_uri" //
155     );
156     final Boolean stackTrace = this.stackTraceIndicator.stackTrace(null, null);
157     final String path = requestUri != null ? requestUri.toString() : null;
158     // TODO final String track = stackTrace ? curl(request) : TRACE_OFF; ? 忘记为什么这里简单使用requestUri
159     final String track = stackTrace ? path : TRACE_OFF;
160 
161     return this.resolve(requestAttributes, throwable, now, stackTrace, path, track);
162   }

164   protected ResolvedError resolve( //
165     final RequestAttributes requestAttributes, //
166     final T throwable, //
167     final DateTime now, //
168     final Boolean stackTrace, //
169     final String path, //
170     final String track //
171   ) {
172     final ResolvedError resolvedError;
173     if (isResolvedError(throwable)) {
174       resolvedError = ((ResolvedErrorException) throwable).getError();
175       resolvedError.trackPrepend(track);
176     } else if (isResolvedErrorWrapByOther(throwable)) {
177       // 注意顺序
178       // 1.判断是不是ResolvedErrorException
179       // 2.判断是不是被封装了的ResolvedErrorException(如HystrixException)
180       log.warn("这里将被封装过ResolvedErrorException提取出来，原异常信息为：", throwable);
181       resolvedError = ((ResolvedErrorException) throwable.getCause()).getError();
182       resolvedError.trackPrepend(track);
183     } else {
184       final Location location = find(throwable).orElse(null);
185       // basic
186       final Optional<List<ValidationError>> errorsOptional = this.validationErrors(throwable);
187       final Optional<Integer> statusOptional = this.status(requestAttributes, location, throwable);
188       final String error = error(statusOptional);
189       final List<ValidationError> errors = errorsOptional.orElse(null);
190       final String exception = throwable != null ? throwable.getClass().getName() : null;
191       final String message = message(requestAttributes, throwable, errorsOptional);
192       final Integer status = statusOptional.orElse(500);
193       final Long timestamp = now.getMillis();
194       final String trace = stackTrace && throwable != null ? getStackTraceAsString(throwable) : null;
195       // extended
196       final String datetime = now.toString(Defaults.ISO8601);
197       final String localizedMessage = this.localizedMessage(requestAttributes, location, throwable).orElse("null");
198       final HttpHeaders headers = this.createHeaders(requestAttributes, throwable).orElse(null);
199       final List<String> tracks = ImmutableList.of(track);
200 
201       resolvedError = ResolvedError.resolvedErrorBuilder() //
202         // basic
203         .error(error) //
204         .errors(errors != null ? errors.toArray(new ValidationError[errors.size()]) : null) //
205         .exception(exception) //
206         .message(message) //
207         .path(path) //
208         .status(status) //
209         .timestamp(timestamp) //
210         .trace(trace) //
211         // extended
212         .datetime(datetime) //
213         .localizedMessage(localizedMessage) //
214         .headers(HttpHeader.fromHttpHeaders(headers)) //
215         .tracks(tracks.toArray(new String[tracks.size()])) //
216         .build();
217     }
218 
219     logError(requestAttributes, throwable, resolvedError);
220 
221     return resolvedError;
222   }

224   /**
225    * Logs the exception; on ERROR level when status is 5xx, otherwise on INFO level without stack
226    * trace, or DEBUG level with stack trace. The logger name is {@code ExceptionResolver}.
227    *
228    * @param requestAttributes requestAttributes
229    * @param throwable         throwable
230    * @param resolvedError     The exception to log.
231    */
232   protected void logError( //
233     final RequestAttributes requestAttributes, //
234     final T throwable, //
235     final ResolvedError resolvedError //
236   ) {
237     if (resolvedError.getStatus() >= HttpStatus.INTERNAL_SERVER_ERROR.value()) {
238       final Marker marker = MarkerFactory.getMarker("error");
239       final String msg = String.format( //
240         "%s ~> %d", //
241         resolvedError.getPath(), //
242         resolvedError.getStatus() //
243       );
244 
245       if (log.isTraceEnabled()) {
246         log.trace("attributes in request scope: {}", //
247           newArrayList(requestAttributes.getAttributeNames(SCOPE_REQUEST)));
248       }
249       log.warn(marker, msg, new Object[]{throwable});
250     }
