16  @Slf4j
17  public class OnNotEnvProductionCondition extends SpringBootCondition {
18  
19    @Override
20    public ConditionOutcome getMatchOutcome( //
21        final ConditionContext context, //
22        final AnnotatedTypeMetadata metadata //
23    ) {
24      final MultiValueMap<String, Object> allAnnotationAttributes = metadata.getAllAnnotationAttributes( //
25          ConditionalOnNotEnvProduction.class.getName(), true //
26      );
27  
28      final String environment = AppProperties.getEnvironment(context.getEnvironment());
29      final Boolean prodEnvironment = AppProperties.getProdEnvironment(environment);
30      final ConditionOutcome outcome;
31      if (allAnnotationAttributes != null && prodEnvironment) {
32        outcome = ConditionOutcome.noMatch("Environment: " + environment + " not match, is production environment.");
33      } else {
34        outcome = ConditionOutcome.match();
35      }
36      return outcome;
37    }
38  }
