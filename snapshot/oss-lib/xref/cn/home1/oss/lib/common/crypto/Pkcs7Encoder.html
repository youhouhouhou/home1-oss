3   import static java.nio.charset.StandardCharsets.UTF_8;
4   
5   import java.nio.charset.Charset;
6   import java.util.Arrays;
7   
8   /**
9    * 提供基于PKCS7算法的加解密接口.
10   */
11  public enum Pkcs7Encoder {
12  
13    PKCS7_UTF8_BLOCK32(UTF_8, 32);
14  
15    private final Charset charset;
16    private final int blockSize;
17  
18    Pkcs7Encoder(final Charset charset, final int blockSize) {
19      this.charset = charset;
20      this.blockSize = blockSize;
21    }
22  
23    /**
24     * 将数字转化成ASCII码对应的字符，用于对明文进行补码.
25     *
26     * @param value 需要转化的数字
27     * @return 转化得到的字符
28     */
29    private static char chr(final int value) {
30      return (char) (byte) (value & 0xFF);
31    }
32  
33    /**
34     * 获得对明文进行补位填充的字节.
35     *
36     * @param count 需要进行填充补位操作的明文字节个数
37     * @return 补齐用的字节数组
38     */
39    public byte[] encode(final int count) {
40      // 计算需要填充的位数
41      final int toPad = this.blockSize - (count % this.blockSize);
42      final int amountToPad = toPad == 0 ? this.blockSize : toPad;
43      // 获得补位所用的字符
44      final char padChr = Pkcs7Encoder.chr(amountToPad);
45      final StringBuilder sb = new StringBuilder();
46      for (int index = 0; index < amountToPad; index++) {
47        sb.append(padChr);
48      }
49      return sb.toString().getBytes(this.charset);
50    }
51  
52    /**
53     * 删除解密后明文的补位字符.
54     *
55     * @param decrypted 解密后的明文
56     * @return 删除补位字符后的明文
57     */
58    public byte[] decode(final byte[] decrypted) {
59      final int pad = (int) decrypted[decrypted.length - 1];
60      final int padAmount = pad < 1 || pad > this.blockSize ? 0 : pad;
61      return Arrays.copyOfRange(decrypted, 0, decrypted.length - padAmount);
62    }
63  }
