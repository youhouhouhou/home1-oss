5   import lombok.SneakyThrows;

7   import org.apache.commons.io.IOUtils;
8   import org.springframework.core.io.ClassPathResource;
9   
10  /**
11   * Created by zhanghaolun on 16/7/9.
12   */
13  public final class ResourceUtils {
14  
15    private ResourceUtils() {
16    }
17  
18    @SneakyThrows
19    public static String classPathString(final String file) {
20      return IOUtils.toString(classPathBytes(file), UTF_8.name());
21    }
22  
23    @SneakyThrows
24    public static byte[] classPathBytes(final String file) {
25      return IOUtils.toByteArray(
26        new ClassPathResource(file, Thread.currentThread().getContextClassLoader()).getInputStream());
27    }
28  }
