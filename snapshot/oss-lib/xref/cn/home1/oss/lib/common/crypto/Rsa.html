13  
14    @Getter

16    @Getter
17    private final RsaDecryptor decryptor;
18  
19    public Rsa(final RsaEncryptor encryptor, final RsaDecryptor decryptor) {

21      this.decryptor = decryptor;
22    }
23  
24    public Rsa(final Provider provider, final KeyExpression keyExpression) {
25      this( //
26        new RsaEncryptor(provider, keyExpression), //
27        new RsaDecryptor(provider, keyExpression) //
28      );
29    }
30  
31    @Override
32    public String encrypt(final String plainText) {
33      return this.encryptor.encrypt(plainText);
34    }
35  
36    @Override
37    public String encrypt(final String plainText, final Integer maxAge) {
38      return this.encryptor.encrypt(plainText, maxAge);
39    }
40  
41    @Override
42    public String decrypt(final String encryptedAndEncoded) {
43      return this.decryptor.decrypt(encryptedAndEncoded);
44    }
