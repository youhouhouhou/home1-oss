35    default Optional<Class<?>> findClass(final String className) {
36      Class<?> classFound;
37      try {
38        classFound = Class.forName(className);
39      } catch (final ClassNotFoundException ex) {
40        LoggerFactory.getLogger(Jackson2Configurator.class).debug("{} not found", className, ex);
41        classFound = null;
42      }
43      return Optional.ofNullable(classFound);
44    }
45  
46    default Optional<String> getProperty(final PropertyResolver propertyResolver, final String key) {
47      final Optional<String> result;
48      if (propertyResolver != null) {
49        final String property = propertyResolver.getProperty(key);
50        result = Optional.ofNullable(property);
51      } else {
52        result = Optional.empty();
53      }
54      return result;
55    }
56  
57    default <M extends ObjectMapper> Boolean isXmlMapper(final M mapper) {
58      final Optional<Class<?>> optional = findClass(XMLMAPPER_CLASSNAME);
59      return optional.map(xmlMapperClass -> xmlMapperClass.isAssignableFrom(mapper.getClass())).orElse(FALSE);
60    }

80          mapper.configure(JsonParser.Feature.ALLOW_NON_NUMERIC_NUMBERS, true);
81          return mapper;

85        @Override
86        public <M extends ObjectMapper> M config(final PropertyResolver propertyResolver, final M mapper) {
87          mapper.setTimeZone(Defaults.UTC_P8.toTimeZone());
88          mapper.registerModule(new JodaModule());
89          // Jdk8Module ?

99      JACKSON2_HAL_CONFIGURATOR {
100 
101       private static final String MODULE_CLASS = "org.springframework.hateoas.hal.Jackson2HalModule";
102 
103       @Override
104       public <M extends ObjectMapper> M config(final PropertyResolver propertyResolver, final M mapper) {
105         final Optional<Class<?>> moduleClass = this.findClass(MODULE_CLASS);
106         if (moduleClass.isPresent()) {
107           // need HalHandlerInstantiator or lead to exception on data-rest request
108           final Boolean isAlreadyRegisteredIn = this.isAlreadyRegisteredIn(mapper, moduleClass.get());
109           if (!isAlreadyRegisteredIn) {
110             try {
111               mapper.registerModule((Module) moduleClass.get().newInstance());
112             } catch (final ReflectiveOperationException ex) {
113               log.info("Jackson2HalModule config error", ex);
114             }
115           }
116         }
117         return mapper;
118       }
119 
120       <M extends ObjectMapper> Boolean isAlreadyRegisteredIn(final M mapper, final Class<?> jackson2HalModuleClass) {
121         Boolean result;
122         try {
123           final Method isAlreadyRegisteredIn = jackson2HalModuleClass.getDeclaredMethod( //
124             "isAlreadyRegisteredIn", ObjectMapper.class);
125           result = (Boolean) isAlreadyRegisteredIn.invoke(null, mapper);
126         } catch (final ReflectiveOperationException | SecurityException | IllegalArgumentException ex) {
127           log.info("Jackson2HalModule config error", ex);
128           result = FALSE;
129         }
130         return result;
131       }

137     JACKSON2_JAXB_ANNOTATION_CONFIGUATOR {
138 
139       private static final String MODULE_CLASS = "com.fasterxml.jackson.module.jaxb.JaxbAnnotationModule";
140       private static final String JAXB_ANNOTATION_INTROSPECTOR_CLASS = //
141         "com.fasterxml.jackson.module.jaxb.JaxbAnnotationIntrospector";
142 
143       @Override
144       public <M extends ObjectMapper> M config(final PropertyResolver propertyResolver, final M mapper) {
145         final Boolean isXmlMapper = this.isXmlMapper(mapper);
146         final String jaxbEnabledDefault = isXmlMapper ? "true" : "false";
147         final Boolean jaxbEnabled = parseBoolean( //
148           this.getProperty(propertyResolver, JACKSON_JAXB_ENABLED).orElse(jaxbEnabledDefault));
149 
150         final Optional<Class<?>> moduleClass = this.findClass(MODULE_CLASS);
151         if (jaxbEnabled && moduleClass.isPresent()) {
152           try {
153             final Class<?> jaxbClass = this.findClass(JAXB_ANNOTATION_INTROSPECTOR_CLASS).get();
154             final Module module = (Module) moduleClass.get().getConstructor(jaxbClass) //
155               .newInstance(new Jackson2HackedJaxbAnnotationIntrospector());
156             final Class<?> enumType = this.findClass(MODULE_CLASS + "$Priority").orElse(null);
157 
158             final String priorityName = isXmlMapper ? "PRIMARY" : "SECONDARY";
159             final Object priority = this.getEnumValue(enumType, priorityName);
160             final Method setPriorityMethod = moduleClass.get().getDeclaredMethod("setPriority", enumType);
161             setPriorityMethod.invoke(module, priority);
162 
163             mapper.registerModule(module);
164           } catch (final ReflectiveOperationException ex) {
165             log.info("JaxbAnnotationModule config error", ex);
166           }
167         }
168         return mapper;
169       }
170 
171       private Object getEnumValue(final Class<?> enumType, final String name) throws ReflectiveOperationException {
172         final Object result;
173         if (enumType != null) {
174           final Method method = enumType.getDeclaredMethod("valueOf", String.class);
175           result = method.invoke(enumType, name);
176         } else {
177           result = null;
178         }
179         return result;
180       }
