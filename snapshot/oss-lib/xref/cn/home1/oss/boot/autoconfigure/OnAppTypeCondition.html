5   import org.apache.commons.lang3.ArrayUtils;
6   import org.springframework.boot.autoconfigure.condition.ConditionOutcome;

8   import org.springframework.context.annotation.ConditionContext;

10  import org.springframework.core.type.AnnotatedTypeMetadata;
11  import org.springframework.util.MultiValueMap;
12  
13  @Slf4j
14  public class OnAppTypeCondition extends SpringBootCondition {
15  
16    public static Boolean matches(final AppType appType, final AppType... appTypes) {
17      return ArrayUtils.contains(appTypes, appType);
18    }
19  
20    @Override
21    public ConditionOutcome getMatchOutcome( //
22      final ConditionContext context, //
23      final AnnotatedTypeMetadata metadata //
24    ) {
25      final MultiValueMap<String, Object> attributes = metadata.getAllAnnotationAttributes( //
26        ConditionalOnAppType.class.getName(), true //
27      );
28      final AppType[] value = (AppType[]) attributes.get("value").get(0);
29  
30      final Environment environment = context.getEnvironment();
31      final String key = "app.type";
32      final AppType appType = AppType.valueOf(environment.getProperty(key, AppProperties.DEFAULT_APP_TYPE.name()).toUpperCase());
33      final Boolean matches = OnAppTypeCondition.matches(appType, value);
34  
35      final ConditionOutcome outcome;
36      if (matches) {
37        outcome = ConditionOutcome.match();
38      } else {
39        outcome = ConditionOutcome.noMatch( //
40          key + " not match. current: " + appType + ", condition: " + ArrayUtils.toString(value) //
41        );
42      }
43      return outcome;
44    }
45  }
