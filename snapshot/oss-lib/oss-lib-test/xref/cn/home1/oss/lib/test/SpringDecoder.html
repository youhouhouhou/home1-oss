22  import java.lang.reflect.Type;

24  /*
25   * @author Spencer Gibb
26   */
27  public class SpringDecoder implements Decoder {
28  
29    private final ObjectFactory<HttpMessageConverters> messageConverters;
30  
31    public SpringDecoder(final ObjectFactory<HttpMessageConverters> messageConverters) {
32      this.messageConverters = messageConverters;
33    }
34  
35    @Override
36    public Object decode(final Response response, final Type type) throws IOException, FeignException {
37      if (type instanceof Class || type instanceof ParameterizedType) {
38        @SuppressWarnings({"unchecked", "rawtypes"})
39        final HttpMessageConverterExtractor<?> extractor =
40          new HttpMessageConverterExtractor(type, this.messageConverters.getObject().getConverters());
41  
42        return extractor.extractData(new FeignResponseAdapter(response));
43      }
44      throw new DecodeException("type is not an instance of Class or ParameterizedType: " + type);
45    }
46  
47  
48    @Slf4j
49    private static class FeignResponseAdapter implements ClientHttpResponse {
50  
51      private final Response response;
52  
53      FeignResponseAdapter(final Response response) {
54        this.response = response;
55      }
56  
57      @Override
58      public HttpStatus getStatusCode() throws IOException {
59        return HttpStatus.valueOf(this.response.status());
60      }
61  
62      @Override
63      public int getRawStatusCode() throws IOException {
64        return this.response.status();
65      }
66  
67      @Override
68      public String getStatusText() throws IOException {
69        return this.response.reason();
70      }
71  
72      @Override
73      public void close() {
74        try {
75          this.response.body().close();
76        } catch (final IOException ex) {
77          // Ignore exception on close...
78          log.debug("error close feign response body.", ex);
79        }
80      }
81  
82      @Override
83      public InputStream getBody() throws IOException {
84        return this.response.body().asInputStream();
85      }
86  
87      @Override
88      public HttpHeaders getHeaders() {
89        final HttpHeaders responseHeaders = new HttpHeaders();
90        response.headers().entrySet().stream() //
91          .forEach(entry -> responseHeaders.put(entry.getKey(), Lists.newArrayList(entry.getValue())));
92        return responseHeaders;
93      }
94    }
95  }
