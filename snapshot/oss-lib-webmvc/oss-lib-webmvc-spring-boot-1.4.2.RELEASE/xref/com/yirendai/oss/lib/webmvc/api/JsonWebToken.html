12  public class JsonWebToken<T> implements TypeSafeToken<T> {
13  
14    private final TypeSafeToken<T> token;
15    @Getter
16    private final Jwt jwt;
17    /**
18     * Maximum age of the jwt token in seconds.
19     */
20    @Getter
21    private final Integer maxAge;
22  
23    public JsonWebToken( //
24      final TypeSafeToken<T> token, //
25      final Jwt jwt, //
26      final int maxAge //
27    ) {
28      this.token = checkNotNull(token);
29      this.jwt = jwt;
30      this.maxAge = maxAge > 0 ? maxAge : 0;
31    }
32  
33    @Override
34    public String toToken(final T obj) {
35      final String token = this.token.toToken(obj);
36      return this.jwt.encrypt(token, this.maxAge);
37    }
38  
39    @Override
40    public T fromToken(final String compactJws) {
41      final String token = this.jwt.decrypt(compactJws);
42      return this.token.fromToken(token);
43    }
44  }
