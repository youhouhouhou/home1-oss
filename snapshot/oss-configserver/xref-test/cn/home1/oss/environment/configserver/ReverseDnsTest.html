26    /**
27     * REVERSE DNS LOOKUP USING JNDI
28     * In this example the IP being looked up is 10.141.4.186 The octets are reversed (186.4.141.10)
29     * and appended to the in-addr.arpa zone: 186.4.141.10.in-addr.arpa
30     */
31    public static void reverseJndi() {
32      try {
33        final Hashtable<String, String> env = new Hashtable<>();
34        env.put("java.naming.factory.initial", "com.sun.jndi.dns.DnsContextFactory");
35  
36        final DirContext ctx = new InitialDirContext(env);
37        //final Attributes attrs = ctx.getAttributes("186.4.141.10", new String[]{"PTR"});
38        final Attributes attrs = ctx.getAttributes("186.4.141.10.in-addr.arpa", new String[]{"PTR"});
39  
40        for (final NamingEnumeration ae = attrs.getAll(); ae.hasMoreElements(); ) {
41          final Attribute attr = (Attribute) ae.next();
42          final String attrId = attr.getID();
43          for (final Enumeration vals = attr.getAll(); vals.hasMoreElements();
44               System.out.println(attrId + ": " + vals.nextElement()))
45            ;
46        }
47        ctx.close();
48      } catch (Exception e) {
49        log.info("NO REVERSE DNS");
50      }
51    }
