10  public abstract class Security {
11  
12    public static final String USER_USER = "user";
13    public static final String USER_WEBHOOK = "webhook";
14    public static final String ADMIN = "ADMIN";
15    public static final String USER = "USER";
16    public static final String WEBHOOK = "WEBHOOK";
17    public static final String ROLE_ADMIN = "ROLE_ADMIN";
18    public static final String ROLE_USER = "ROLE_USER";
19    public static final String ROLE_WEBHOOK = "ROLE_WEBHOOK";
20    public static final String DELIMITER = "_-_";
21  
22    private Security() {
23    }
24  
25    public static String modifyUsername(final String application, final String username) {
26      checkArgument(!isAnyBlank(application, username), "application or username should not blank");
27  
28      return username + DELIMITER + application;
29    }
30  
31    public static String restoreUsername(final String modifiedUsername) {
32      checkArgument(isNotBlank(modifiedUsername), "username should not blank");
33  
34      final String[] splited = modifiedUsername.split(DELIMITER);
35      return splited.length > 1 ? splited[1] : modifiedUsername;
36    }
37  }
