3   import static org.apache.commons.lang3.StringUtils.isNotBlank;
4   
5   import org.springframework.beans.factory.annotation.Autowired;
6   import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
7   import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
8   import org.springframework.boot.context.properties.EnableConfigurationProperties;
9   import org.springframework.cloud.config.server.config.ConfigServerProperties;
10  import org.springframework.cloud.config.server.config.EnvironmentRepositoryConfiguration;
11  import org.springframework.cloud.config.server.environment.EnvironmentRepository;
12  import org.springframework.context.annotation.Bean;
13  import org.springframework.context.annotation.Configuration;
14  import org.springframework.core.env.ConfigurableEnvironment;
15  
16  @Configuration
17  @ConditionalOnMissingBean(EnvironmentRepository.class)
18  @ConditionalOnProperty(prefix = "spring.cloud.config.server.commonConfig", name = "enabled", havingValue = "true")
19  public class CommonConfigConfiguration extends EnvironmentRepositoryConfiguration {
20  
21    @Configuration
22    @ConditionalOnMissingBean(EnvironmentRepository.class)
23    @ConditionalOnProperty(prefix = "spring.cloud.config.server.commonConfig.git", name = "uri")
24    @EnableConfigurationProperties(value = {ConfigServerProperties.class, CommonConfigProperties.class})
25    public static class GitRepositoryConfiguration {
26  
27      @Autowired
28      private ConfigurableEnvironment environment;
29  
30      @Autowired
31      private ConfigServerProperties configServerProperties;
32  
33      @Autowired
34      private CommonConfigProperties commonConfigProperties;
35  
36      @Bean
37      @ConditionalOnMissingBean(EnvironmentRepository.class)
38      public EnvironmentRepository environmentRepository() {
39        // common-config
40        final String label = this.commonConfigProperties.getLabel();
41        final CommonConfigRepository commonConfigRepository = this.commonConfigProperties.getGit();
42        commonConfigRepository.setApplication(this.commonConfigProperties.getApplication());
43        commonConfigRepository.setLabel(isNotBlank(label) ? label : commonConfigRepository.getDefaultLabel());
44  
45        // multiple
46        final CommonConfigSupportedMultipleJGitEnvironmentRepository repository =
47            new CommonConfigSupportedMultipleJGitEnvironmentRepository(this.environment);
48        repository.setCommonConfigRepository(commonConfigRepository);
49        if (this.configServerProperties.getDefaultLabel() != null) {
50          repository.setDefaultLabel(this.configServerProperties.getDefaultLabel());
51        }
52  
53        return repository;
54      }
55    }
56  }
