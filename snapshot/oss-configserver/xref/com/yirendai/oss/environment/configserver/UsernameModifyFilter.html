46    public static final Pattern PATH_PATTERN = Pattern.compile("^/([^/]+)/([^/]+)(/([^/]+)?)?");
47    public static final Pattern FILE_PATTERN = Pattern.compile("^(/[^/]+)?/(.+)-(.+)\\.(json|properties|yaml|yml)");

64      this.managementRequestMatcher = request -> request != null //
65          ? isManagementPath(managementContextPath, request.getServletPath()) : false;

68    @Override
69    public void doFilter( //
70        final ServletRequest req, //
71        final ServletResponse res, //
72        final FilterChain chain //
73    ) throws IOException, ServletException {
74      final HttpServletRequest request = (HttpServletRequest) req;
75      final HttpServletResponse response = (HttpServletResponse) res;
76  
77      final String[] tokens = this.tokens(request);
78  
79      if (this.loginRequestMatcher != null && this.loginRequestMatcher.matches(request) && tokens != null) {
80        final String applicationOrUsername = tokens[0];
81  
82        if (this.adminUsername.equals(applicationOrUsername)) { // admin
83          chain.doFilter(request, response);
84        } else {
85          final String name = Security.modifyUsername(applicationOrUsername, Security.USER_USER);
86          final String headerValue = BasicAuthUtils.basicAuthHeader(name, tokens[1]);
87          chain.doFilter(new AuthorizationHeaderWrapper(request, headerValue), response);
88        }
89      } else if (!this.managementRequestMatcher.matches(request) && tokens != null) {
90        final ConfigCoords configCoords = this.extractConfigCoords(request);
91        final String application = configCoords.getApplication();
92  
93        final String username = tokens[0];
94  
95        if (Security.USER_USER.equals(username) && isNotBlank(application)) {
96          final String name = Security.modifyUsername(application, Security.USER_USER);
97          final String headerValue = BasicAuthUtils.basicAuthHeader(name, tokens[1]);
98          chain.doFilter(new AuthorizationHeaderWrapper(request, headerValue), response);
99        } else { // admin
100         chain.doFilter(request, response);
101       }
102     } else {
103       chain.doFilter(request, response);
104     }
105   }

107   private String[] tokens(final HttpServletRequest request) {
108     final String header = request.getHeader(BasicAuthUtils.BASIC_AUTH_HEADE_NAME);
109     final boolean hasBasicAuthHeader = header != null && header.startsWith("Basic ");
110     final String[] tokens;
111     if (hasBasicAuthHeader) {
112       tokens = BasicAuthUtils.extractAndDecodeAuthHeader(header, this.getCredentialsCharset(request));
113       checkArgument(tokens.length == 2, "invalid basic auth header");
114     } else {
115       tokens = null;
116     }
117     return tokens;
118   }

120   public void setCredentialsCharset(final String credentialsCharset) {
121     Assert.hasText(credentialsCharset, "credentialsCharset cannot be null or empty");
122     this.credentialsCharset = credentialsCharset;
123   }

125   protected String getCredentialsCharset(final HttpServletRequest httpRequest) {
126     return this.credentialsCharset != null ? this.credentialsCharset : httpRequest.getCharacterEncoding();
127   }

129   private ConfigCoords extractConfigCoords(final HttpServletRequest request) {
130     final String servletPath = request.getServletPath();
131     final String path = isNotBlank(this.configServerPrefix) ? //
132         replaceOnce(servletPath, this.configServerPrefix, "") : //
133         servletPath;
134     return extractConfigCoords(path);
135   }
136 
137   /**
138    * /{application}/{profile}[/{label}]
139    * /{application}-{profile}.yml
140    * /{label}/{application}-{profile}.yml
141    * /{application}-{profile}.properties
142    * /{label}/{application}-{profile}.properties
143    */
144   public static ConfigCoords extractConfigCoords(final String path) {
145     final Matcher fileMatcher = FILE_PATTERN.matcher(path);
146     final Matcher pathMatcher = PATH_PATTERN.matcher(path);
147 
148     final String application;
149     final String profile;
150     final String label;
151     final String ext;
152     if (fileMatcher.matches()) {
153       label = fileMatcher.group(1) != null ? fileMatcher.group(1).substring(1) : ""; // could be null if not found
154       application = fileMatcher.group(2);
155       profile = fileMatcher.group(3);
156       ext = fileMatcher.group(4);
157     } else if (pathMatcher.matches()) { // pathMatcher.find()
158       application = pathMatcher.group(1);
159       profile = pathMatcher.group(2);
160       label = pathMatcher.group(3) != null ? pathMatcher.group(3).substring(1) : "";
161       ext = "";
162     } else {
163       application = "";
164       profile = "";
165       label = "";
166       ext = "";
167     }
168     return new ConfigCoords(application, label, profile, ext);
169   }
