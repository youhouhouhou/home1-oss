21  /**
22   * Reverse DNS resolve.
23   * see: http://archive.oreilly.com/pub/post/reverse_dns_lookup_and_java.html
24   * Created by zhanghaolun on 16/11/2.
25   */
26  @NoArgsConstructor(access = PRIVATE)
27  public abstract class ReverseDns {
28  
29    public static String reverseSystemDns(final String hostIp) throws UnknownHostException {
30      final InetAddress addr = InetAddress.getByName(hostIp);
31      return addr.getHostName(); // getCanonicalHostName()
32    }
33  
34    public static String reverseJavaDns(final String hostIp) throws IOException {
35      //final Record opt = null;
36      final Resolver resolver = new ExtendedResolver();
37      final Name name = ReverseMap.fromAddress(hostIp);
38      final int type = Type.PTR;
39      final int dclass = DClass.IN;
40      final Record rec = Record.newRecord(name, type, dclass);
41      final Message query = Message.newQuery(rec);
42  
43      final Message response = resolver.send(query);
44      final Record[] answers = response.getSectionArray(Section.ANSWER);
45  
46      final String result;
47      if (answers.length == 0) {
48        result = hostIp;
49      } else {
50        result = answers[0].rdataToString();
51      }
52      return result;
53    }
54  }
