5   import org.springframework.beans.factory.annotation.Autowired;
6   import org.springframework.boot.context.properties.ConfigurationProperties;
7   import org.springframework.cloud.config.environment.Environment;
8   import org.springframework.cloud.config.server.environment.MultipleJGitEnvironmentRepository;
9   import org.springframework.core.env.ConfigurableEnvironment;
10  
11  @ConfigurationProperties("spring.cloud.config.server.git")
12  public class CommonConfigSupportedMultipleJGitEnvironmentRepository extends MultipleJGitEnvironmentRepository {
13  
14    @Autowired
15    private UserService userService;
16    @Setter
17    private CommonConfigRepository commonConfigRepository;
18  
19    public CommonConfigSupportedMultipleJGitEnvironmentRepository(final ConfigurableEnvironment environment) {
20      super(environment);
21    }
22  
23    public void setUserService(final UserService userService) {
24      synchronized (this) {
25        this.userService = userService;
26      }
27    }
28  
29    @Override
30    public void afterPropertiesSet() throws Exception {
31      super.afterPropertiesSet();
32      if (this.commonConfigRepository != null) {
33        this.commonConfigRepository.setEnvironment(getEnvironment());
34      }
35    }
36  
37    /**
38     * check permission then read common config.
39     */
40    @Override
41    public Environment findOne(final String application, final String profile, final String label) {
42      synchronized (this) {
43        final boolean authenticated = this.userService.authenticateUser(application);
44  
45        final Environment result;
46        if (authenticated) {
47          final Environment applicationConfig = super.findOne(application, profile, label);
48          final Environment commonConfig = this.commonConfigRepository.findOne(profile);
49          //commonConfig.setLabel(applicationConfig.getLabel());
50          //commonConfig.setName(applicationConfig.getName());
51          //commonConfig.setProfiles(applicationConfig.getProfiles());
52          //commonConfig.setVersion(applicationConfig.getVersion());
53          //commonConfig.getPropertySources().addAll(applicationConfig.getPropertySources());
54          // application config first
55          applicationConfig.getPropertySources().addAll(commonConfig.getPropertySources());
56          result = applicationConfig;
57        } else {
58          result = new Environment(application, profile);
59        }
60  
61        return result;
62      }
63    }
64  }
