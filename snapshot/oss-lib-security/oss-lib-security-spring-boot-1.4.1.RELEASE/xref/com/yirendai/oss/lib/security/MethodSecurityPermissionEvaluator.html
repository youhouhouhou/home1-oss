19  @Slf4j
20  public class MethodSecurityPermissionEvaluator implements PermissionEvaluator {
21  
22    public static final String HAS_PERMISSION_3 =
23      "permissionEvaluator hasPermission(authentication, targetDomainObject, permission)";
24    public static final String HAS_PERMISSION_4 =
25      "permissionEvaluator hasPermission(authentication, targetId, targetType, permission)";
26  
27    private Boolean securityEnabled;
28  
29    private PermissionEvaluator projectPermissionEvaluator;
30    @SuppressWarnings("rawtypes")
31    private DiscoverableEnums<? extends StaticPrivilege> staticPrivileges;
32  
33    @SuppressWarnings("unchecked")
34    public void setProjectPermissionEvaluator(final PermissionEvaluator projectPermissionEvaluator) {
35      final String basePackage = AppUtils.appBasePackage(projectPermissionEvaluator.getClass().getName().split("\\.")[0]);
36      this.staticPrivileges = DiscoverableEnums.discoverableEnums(basePackage, StaticPrivilege.class);
37      this.projectPermissionEvaluator = projectPermissionEvaluator;
38    }
39  
40    boolean hasStaticPermission( //
41      final GenericUser genericUser, //
42      final String targetDomainObject, //
43      final String permission //
44    ) {
45      final String privilege = StaticPrivilege.toAuthority(targetDomainObject, permission);
46      @SuppressWarnings("rawtypes")
47      final StaticPrivilege found = this.staticPrivileges.convert(privilege);
48  
49      return genericUser != null && //
50        genericUser.getAuthorities() != null && //
51        genericUser.getAuthorities().contains(new BaseGrantedAuthority(found.getAuthority()));
52    }
53  
54    @Override
55    public boolean hasPermission(final Authentication authentication, final Object targetDomainObject,
56      final Object permission) {
57      final boolean hasPermission;
58      if (!this.securityEnabled) {
59        log.info("{} securityEnabled not enabled, returning true", HAS_PERMISSION_3);
60        hasPermission = true;
61      } else if (authentication == null) {
62        log.warn("{} authentication is null, returning false", HAS_PERMISSION_3);
63        hasPermission = false;
64      } else if (targetDomainObject == null) {
65        log.warn("{} targetDomainObject is null, returning false", HAS_PERMISSION_3);
66        hasPermission = false;
67      } else if (permission == null) {
68        log.warn("{} permission is null, returning false", HAS_PERMISSION_3);
69        hasPermission = false;
70      } else {
71        final GenericUser genericUser = GenericUser.fromPrincipal(authentication);
72        if (String.class.isAssignableFrom(targetDomainObject.getClass())
73          && String.class.isAssignableFrom(permission.getClass())) {
74          hasPermission = this.hasStaticPermission(genericUser, targetDomainObject.toString(), permission.toString());
75        } else {
76          hasPermission = this.projectPermissionEvaluator.hasPermission(authentication, targetDomainObject, permission);
77        }
78      }
79      return hasPermission;
80    }
81  
82    @Override
83    public boolean hasPermission(final Authentication authentication, final Serializable targetId,
84      final String targetType, final Object permission) {
85      final boolean hasPermission;
86      if (!this.securityEnabled) {
87        log.info("{} securityEnabled not enabled, returning true", HAS_PERMISSION_4);
88        hasPermission = true;
89      } else {
90        // Not required here.
91        log.warn("{} not implemented, returning false", HAS_PERMISSION_4);
92        hasPermission = false;
93      }
94      return hasPermission;
95    }
96  
97    public void setSecurityEnabled(final Boolean securityEnabled) {
98      this.securityEnabled = securityEnabled;
99    }
100 }
