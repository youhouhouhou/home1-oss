23  abstract class OAuth2Utils {
24  
25    private OAuth2Utils() {
26    }
27  
28    static Boolean isOAuth2Authentication(final Principal principal) {
29      return principal instanceof OAuth2Authentication;
30    }
31  
32    static GenericUser fromOAuth2Authentication(final Principal principal) {
33      final OAuth2Authentication oauth2Authentication = (OAuth2Authentication) principal;
34  
35      final Authentication auth = oauth2Authentication.getUserAuthentication();
36      if (oauth2Authentication.isClientOnly() || auth == null || auth.getDetails() == null) {
37        return null;
38      }
39  
40      final boolean accountNonExpired = true;
41      final boolean accountNonLocked = true;
42      @SuppressWarnings("unchecked")
43      final Map<String, Object> details = (Map<String, Object>) auth.getDetails();
44      @SuppressWarnings("unchecked")
45      final Set<GrantedAuthority> authorities =
46        ((Collection<Map<String, String>>) details.get("authorities")).stream()
47          .map(authority -> new BaseGrantedAuthority(authority.get("authority")))
48          .collect(toSet());
49      final boolean credentialsNonExpired = true;
50      final boolean enabled = true;
51      final String password = System.getProperty("blankPassword", "");
52      final String username = auth.getPrincipal().toString();
53      final String uuid = ((OAuth2AuthenticationDetails) oauth2Authentication.getDetails()).getTokenValue();
54  
55      return new GenericUser(accountNonExpired, accountNonLocked, authorities, credentialsNonExpired,
56        enabled, password, username, ImmutableMap.of(), // TODO pass properties ?
57        Defaults.now(), uuid);
58    }
59  }
