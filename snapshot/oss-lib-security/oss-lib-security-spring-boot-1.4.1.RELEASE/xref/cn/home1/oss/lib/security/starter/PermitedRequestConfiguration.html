66      // ! TODO ! 其它地方似乎也加了这些, 条件开启, 重复代码 see: Swagger2DocumentationAutoConfiguration.swaggerRequestMatcher

73      ignored.add("/h2-console/**");
74      if (this.h2ConsoleProperties != null) {
75        final String path = this.h2ConsoleProperties.getPath();
76        final String antPattern = (path.endsWith("/") ? path + "**" : path + "/**");
77        ignored.add(antPattern);
78      }
79  
80      // create another filterChain after spring-boot's default (security.ignored) filterChain
81      if (!ignored.isEmpty()) {
82        final List<RequestMatcher> matchers = ignored.stream() //
83          .map(pattern -> new AntPathRequestMatcher(pattern, null)) //
84          .collect(toList());
85        final RequestMatcher requestMatcher = new OrRequestMatcher(matchers);
86        web.ignoring().requestMatchers(requestMatcher);
87      }
88    }

95    @Bean(name = PERMITED_REQUEST_MATCHER)
96    public RequestMatcher permitedRequestMatcher() {
97      final RequestMatcher result;
98      if (this.appProperties.getSecurityEnabled()) {
99        // requestMatchers must contain a value
100       final List<RequestMatcher> antMatchers = this.permitedRequests().entrySet().stream()
101         .flatMap(entry -> entry.getValue().stream().map(pattern -> //
102           isBlank(entry.getKey()) //
103             ? new AntPathRequestMatcher(pattern) : new AntPathRequestMatcher(pattern, entry.getKey())
104         )).collect(toList());
105       result = antMatchers.isEmpty() ? request -> false : new OrRequestMatcher(antMatchers);
106     } else {
107       result = AnyRequestMatcher.INSTANCE;
108     }
109     return result;
110   }
111 
112   @Bean(name = PERMITED_REQUESTS)
113   public Map<String, List<String>> permitedRequests() {
114     // permit的还是要过filter和authenticationManager
115     final AppSecurityProperties security = this.appProperties.getSecurity();
116 
117     final String loginPage = security.getLoginPage();
118 
119     final Collection<String> defaultPermited = this.appProperties.getType() != RESOURCE ? //
120       newLinkedHashSet(newArrayList( //
121         security.getLoginPublicKeyUrl(), loginPage, security.getLoginProcessingUrl(), //
122         security.getLogoutUrl() + "/*", security.getLogoutUrl() //
123       )) : newArrayList();
124 
125     final Collection<String> permitedRequests = newLinkedHashSet();
126 
127     if (isNotBlank(security.getPermited())) {
128       permitedRequests.addAll(newArrayList(security.getPermited().split("[ ]*,[ ]*")));
129     }
130 
131     permitedRequests.addAll(defaultPermited);
132 
133     if (this.verifyCodeFilter != null) {
134       permitedRequests.add(this.verifyCodeFilter.getCodeUrl());
135     }
136 
137     final Map<String, List<String>> grouped = permitedRequests.stream()
138       .map(expression -> {
139         final String[] fragments = expression.split(":");
140         return fragments.length == 1 //
141           ? newArrayList("", fragments[0]) : newArrayList(fragments[0].toUpperCase(), fragments[1]);
142       })
143       .collect(groupingBy(o -> o.get(0), mapping(o -> o.get(1), toList())));
144 
145     return ImmutableMap.copyOf(grouped);
146   }
