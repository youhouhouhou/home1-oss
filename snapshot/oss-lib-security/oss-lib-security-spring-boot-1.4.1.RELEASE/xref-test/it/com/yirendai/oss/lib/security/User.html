39  @Entity
40  @Builder(builderMethodName = "userBuilder")
41  @AllArgsConstructor
42  @NoArgsConstructor
43  @Setter(PROTECTED)
44  @Getter
45  public class User extends AbstractUser {
46  
47    private static final long serialVersionUID = 1L;
48  
49    // build in properties
50  
51    @Type(type = "yes_no")
52    private boolean enabled;
53  
54    @Column(nullable = false, length = 80)
55    @NonNull
56    private String name;
57  
58    @Column(nullable = false, length = 80)
59    @NonNull
60    private String password;
61  
62    @NonNull
63    @ManyToMany(targetEntity = UserRole.class, fetch = FetchType.EAGER)
64    private Set<UserRole> roles = new HashSet<>();
65  
66    @Id
67    @GeneratedValue
68    private Long userId;
69  
70    // extended properties
71    @Column(nullable = false, length = 80)
72    @NonNull
73    private String email;
74  
75    public void addRole(final UserRole role) {
76      final Set<UserRole> roles = newLinkedHashSet(this.roles);
77      roles.add(role);
78      this.roles = ImmutableSet.copyOf(roles);
79    }
80  
81    @Override
82    public Set<GrantedAuthority> getAuthorities() {
83      return UserRolePrivileges.getAuthorities(this.getRoles().stream().map(UserRole::getName).collect(toSet()));
84    }
85  
86    @Override
87    public String getId() {
88      return this.userId != null ? this.userId.toString() : null;
89    }
90  
91    @JsonIgnore
92    @Override
93    public Map<String, String> getProperties() {
94      return ImmutableMap.of("email", this.getEmail());
95    }
96  
97    public void passwordEncode(final PasswordEncoder passwordEncoder) {
98      if (passwordEncoder != null && isNotBlank(this.password)) {
99        this.password = passwordEncoder.encode(this.password);
100     }
101   }
102 }
