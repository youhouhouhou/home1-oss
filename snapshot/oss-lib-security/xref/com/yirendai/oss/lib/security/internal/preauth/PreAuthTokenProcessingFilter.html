40  /**
41   * Not a bean, avoid auto pick-up.
42   * after BasicAuthenticationFilter and PreAuthTestUserFilter.
43   * see: BasicAuthenticationFilter or AbstractPreAuthenticatedProcessingFilter
44   *
45   * <p>authenticate against the supplied {@code AuthenticationManager}
46   * and use the supplied {@code AuthenticationEntryPoint} to handle authentication failures.</p>
47   *
48   * @author zhanghaolun
49   */
50  @Setter
51  @Getter(value = PROTECTED)
52  @Slf4j
53  public class PreAuthTokenProcessingFilter extends GenericFilterBean {
54  
55    /**
56     * will be invoked when authentication fails. Typically an instance of {@link
57     * BasicAuthenticationEntryPoint}.
58     */
59    private AuthenticationEntryPoint authenticationEntryPoint;
60    /**
61     * the bean to submit authentication requests to.
62     */
63    private AuthenticationManager authenticationManager;
64    private AuthenticationDetailsSource<HttpServletRequest, ?> authenticationDetailsSource;
65    private RememberMeServices rememberMeServices;
66  
67    public PreAuthTokenProcessingFilter() {
68      this.authenticationEntryPoint = null;
69      this.authenticationManager = new NoOpAuthenticationManager();
70  
71      this.authenticationDetailsSource = new WebAuthenticationDetailsSource();
72      this.rememberMeServices = new NullRememberMeServices();
73    }
74  
75    @Override
76    public void doFilter( //
77      final ServletRequest req, //
78      final ServletResponse res, //
79      final FilterChain chain //
80    ) throws IOException, ServletException {
81      final HttpServletRequest request = (HttpServletRequest) req;
82      final HttpServletResponse response = (HttpServletResponse) res;
83  
84      try {
85        if (this.authenticationIsRequired()) {
86          final GenericUser principal = (GenericUser) request.getAttribute(ATTR_PRINCIPAL);
87          final String token = (String) request.getAttribute(ATTR_PRINCIPAL_TOKEN);
88          request.removeAttribute(ATTR_PRINCIPAL);
89          request.removeAttribute(ATTR_PRINCIPAL_TOKEN);
90  
91          final AbstractAuthenticationToken authRequest = this.attempAuthentication(request, principal, token);
92          authRequest.setDetails(this.authenticationDetailsSource.buildDetails(request));
93  
94          final Authentication authResult = this.authenticationManager.authenticate(authRequest);
95  
96          if (log.isTraceEnabled()) {
97            log.trace("AUTH authentication success: {}, principal: {}", authResult, principal);
98          }
99  
100         final SecurityContext securityContext = SecurityContextHolder.getContext();
101         securityContext.setAuthentication(authResult);
102         this.rememberMeServices.loginSuccess(request, response, authResult);
103         this.onSuccessfulAuthentication(request, response, authResult);
104       }
105     } catch (final AuthenticationException failed) {
106       SecurityContextHolder.clearContext();
107 
108       if (log.isTraceEnabled()) {
109         log.trace("AUTH authentication failed. not login.", failed);
110       }
111 
112       this.rememberMeServices.loginFail(request, response);
113       this.onUnsuccessfulAuthentication(request, response, failed);
114       if (this.getIgnoreFailure()) {
115         chain.doFilter(request, response);
116       } else {
117         this.authenticationEntryPoint.commence(request, response, failed);
118       }
119       return;
120     }
121 
122     chain.doFilter(request, response);
123   }
124 
125   private boolean authenticationIsRequired() {
126     return Security.authenticationIsRequired();
127   }
128 
129   @SuppressWarnings({"squid:S1172"})

133     final String token

136   }

156   public Boolean getIgnoreFailure() {
157     return this.authenticationEntryPoint == null;
158   }
