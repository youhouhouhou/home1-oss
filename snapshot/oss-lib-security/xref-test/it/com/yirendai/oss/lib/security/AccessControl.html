8   import com.yirendai.oss.lib.security.api.GenericUser;
9   
10  import org.springframework.security.core.Authentication;
11  import org.springframework.stereotype.Component;
12  
13  @Component
14  public class AccessControl {
15  
16    public Boolean checkUserId(final Authentication authentication, final Long id) {
17      checkNotNull(authentication, UNAUTHORIZED, "authentication not present");
18      checkNotNull(id, UNAUTHORIZED, "id not present");
19  
20      final Object principal = authentication.getPrincipal();
21      checkNotNull(principal, UNAUTHORIZED, "principal not present");
22      check(principal instanceof GenericUser, UNAUTHORIZED, "unsupported principal type {}", principal.getClass());
23  
24      final GenericUser user = (GenericUser) principal;
25      check(user.getId().equals(id.toString()), UNAUTHORIZED, "user id not match");
26      return TRUE;
27    }
28  }
