26    public static PreAuthenticatedAuthenticationToken authenticated( //
27      final PreAuthenticatedAuthenticationToken authentication //
28    ) {
29      final PreAuthenticatedAuthenticationToken result;
30  
31      if (authentication == null) {
32        result = null;
33      } else {
34        authentication.setAuthenticated(true);
35        authentication.eraseCredentials();
36  
37        final Object principal = authentication.getPrincipal();
38        final Object credentials = authentication.getCredentials();
39  
40        final GenericUser genericUser = isGenericUser(principal) ? (GenericUser) principal : null;
41  
42        if (genericUser == null) {
43          result = authentication;
44        } else {
45          genericUser.eraseCredentials();
46          final Collection<GrantedAuthority> authorities = genericUser.getAuthorities() != null ? //
47            genericUser.getAuthorities() : ImmutableSet.of();
48          result = new PreAuthenticatedAuthenticationToken(principal, credentials, authorities);
49        }
50      }
51      return result;
52    }
53  
54    public static String rawToken(final Authentication authentication, final TypeSafeToken<GenericUser> token) {
55      // TODO find OAuth2Authentication's header Authorization: Bearer?
56  
57      final String result;
58      if (authentication instanceof PreAuthenticatedAuthenticationToken) {
59        final Object principal = authentication.getPrincipal();
60        final GenericUser genericUser = isGenericUser(principal) ? (GenericUser) principal : null;
61        result = genericUser != null ? (String) authentication.getCredentials() : null;
62      } else {
63        final GenericUser genericUser = GenericUser.fromPrincipal(authentication);
64        result = token.toToken(genericUser);
65      }
66      return result;
67    }
68  
69    public static Boolean tokenInSecurityContext() {
70      final SecurityContext securityContext = SecurityContextHolder.getContext();
71      final Authentication authentication = securityContext.getAuthentication();
72      return authentication != null && authentication.isAuthenticated();
73    }
74  
75    public static String tokenFromSecurityContext(final TypeSafeToken<GenericUser> token) {
76      final SecurityContext securityContext = SecurityContextHolder.getContext();
77      final Authentication authentication = securityContext.getAuthentication();
78  
79      final String result;
80      if (authentication != null && authentication.isAuthenticated()) {
81        result = AuthenticationTokens.rawToken(authentication, token);
82      } else {
83        result = null;
84      }
85  
86      return result;
87    }
