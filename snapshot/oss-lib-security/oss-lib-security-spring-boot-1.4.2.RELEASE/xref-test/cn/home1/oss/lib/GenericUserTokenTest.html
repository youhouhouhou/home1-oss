49      TypeSafeToken<GenericUser> token = new JsonToken<>(GenericUser.class, this.objectMapper);

51      final Jwt jwtCipher = Cryptos.cipher(this.appSecurityProperties.getJwtKey());
52      if (jwtCipher != null) {
53        token = new JsonWebToken<>(token, jwtCipher, 3600);

58      final EncodeCipher cookieCipher = Cryptos.cipher(this.appSecurityProperties.getCookieKey());
59      if (cookieCipher != null) {
60        token = new SecureToken<>(token, cookieCipher);
61      }
62  
63      this.token = new UrlEncodedToken<>(token);
64    }

66    @Test
67    public void testGenericUserToken() {
68      final User user = UserDetails.userDetailsBuilder()
69        .authorities(Sets.newHashSet(new BaseGrantedAuthority("ADMIN"))) //
70        .enabled(true) //
71        .id("0") //
72        .name("name") //
73        .password("password") //
74        .properties(ImmutableMap.of()) //
75        .build();
76      final GenericUser genericUser = GenericUser.fromUser(user);
77      final String token = this.token.toToken(genericUser);
78      log.info("token: {}", token);
79      log.info("token.length: {}", token.length());
80      final GenericUser fromToken = this.token.fromToken(token);
81      assertEquals(fromToken, genericUser);
