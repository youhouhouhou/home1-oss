5   import lombok.extern.slf4j.Slf4j;

7   import org.springframework.boot.test.json.AbstractJsonMarshalTester;
8   
9   import java.io.IOException;
10  
11  /**
12   * Created by Meifans on 17/1/5.
13   */
14  @Slf4j
15  public class GenericUserJsonTester {
16  
17    private AbstractJsonMarshalTester<GenericUser> json;
18  
19    public GenericUserJsonTester(AbstractJsonMarshalTester<GenericUser> json) {
20      this.json = json;
21    }
22  
23    public void testGenericUserSerialize() throws IOException {
24      final GenericUser scenario = GenericUserTestScenario.genericUserScenario();
25  
26      this.json.write(scenario).assertThat()
27        .extractingJsonPathMapValue("properties").containsAllEntriesOf(scenario.getProperties());
28  
29      this.json.write(scenario).assertThat()
30        .extractingJsonPathArrayValue("authorities").contains(
31        scenario.getAuthorities().iterator().next().getAuthority());
32  
33      this.json.write(scenario).assertThat()
34        .extractingJsonPathStringValue("timestamp").isEqualTo(ISO8601.print(scenario.getTimestamp()));
35    }
36  
37    public void testGenericUserDeSerialize() throws IOException {
38      this.json.read("/genericUserScenario.json").assertThat().isEqualTo(GenericUserTestScenario.genericUserScenario());
39    }
40  }
